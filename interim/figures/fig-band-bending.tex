\begin{figure}[H]
  \centering
  \begin{tikzpicture}[x=0.9cm,y=0.9cm,>=Stealth,font=\footnotesize]

    % Axes labels
    \node[anchor=north] at (1.5,0) {AlGaAs};
    \node[anchor=north] at (5.5,0) {GaAs};

    % Fermi level reference
    \draw[dotted] (0.2,2.3) -- (6.8,2.3);
    \node[anchor=east] at (0.1,2.3) {$E_F$};

    % Conduction band (left to right with band offset and well)
    \draw (0.2,3.3) -- (3.6,3.3) -- (4.9,1.5) -- (6.6,2.6);
    % Valence band
    \draw (0.2,1.1) -- (3.6,1.1) -- (4.9,-0.2) -- (6.6,1.0);

    % Energy labels on left
    \node[anchor=east] at (0.1,3.3) {$E_C$};
    \node[anchor=east] at (0.1,1.1) {$E_V$};

    % Band step markers (offset illustration)
    \draw[decorate,decoration={brace,amplitude=4pt,mirror}]
      (4.9,1.5) -- (4.9,3.3);
    \node[anchor=west] at (5.05,2.4) {$\Delta E_C$};

    % 2DEG indication at the well
    \draw[->,blue,thick] (5.1,2.9) -- (5.05,2.0);
    \node[blue,anchor=south west] at (5.1,2.9) {2DEG};

    % Donor electron transfer arrow
    \draw[->] (3.0,3.7) .. controls (3.6,3.3) and (4.2,3.0) .. (4.8,2.3);
    \node[anchor=south] at (3.0,3.7) {$e^-$};

    % Region divider
    \draw[dashed] (4.9,-0.2) -- (4.9,3.8);

  \end{tikzpicture}
  \caption{Conduction and valence band alignment at the AlGaAs/GaAs interface showing band bending and the resulting two-dimensional electron gas (2DEG).}
  \label{fig:band_bending}
\end{figure}
